    Export function signature:
    ```typescript
    export function computeConvergenceScore(
      goalSpaceId: string,
      edges: readonly Edge[],
      allNodes: readonly Node[]
    ): ConvergenceResult
    ```

    Implementation algorithm:
    1. Build nodeMap: `new Map(allNodes.map(n => [n.id, n]))`
    2. Find outcome IDs: filter edges where `edge_type === 'advances_goal' && target_id === goalSpaceId`, collect source_ids
    3. If no outcomes, return `{ score: 0, factor_breakdown: { outcome_scores: [], total_outcomes: 0, raw_score: 0 } }`
    4. For each outcome, call `scoreOutcome(outcomeId, outcomeTitle, edges, nodeMap)`:
       a. Find incoming edges: `edges.filter(e => e.target_id === outcomeId)`
       b. Track processed node IDs to avoid double-counting a node
       c. For each incoming edge, look up source node, apply weight rules:
          - `indicates_progress` + promoted -> +3.0
          - `indicates_progress` + human_reviewed -> +2.0
          - `indicates_progress` + other -> +0.5
          - `assigned_to_outcome` -> +2.0
          - `targets_outcome` + hunch + promoted -> +1.0
          - `targets_outcome` + hunch + other -> +0.5
          - `targets_outcome` + intervention -> +1.5
          - Any source with status falsified -> -2.0 (overrides positive from same node)
          - Any source with status suspended -> -1.0 (overrides positive from same node)
       d. If zero `targets_outcome` edges AND zero `assigned_to_outcome` edges -> add -1.0 "no_attention" factor
       e. Return OutcomeScore with positive_factors, negative_factors, summed score
    5. Average: `raw_score = sum(outcome_scores) / outcome_count`
    6. Clamp: `score = Math.max(-10, Math.min(10, raw_score))`
    7. Return ConvergenceResult

    IMPORTANT: When a source node is falsified or suspended, it contributes ONLY the negative weight (not both positive and negative). Check status first; if falsified/suspended, add negative factor and skip positive.

    ```sql
    -- COF OS v0.4 — Convergence Snapshots migration
    -- Stores convergence score time-series per goal space
    -- Run against your Supabase project in the SQL editor or via CLI

    -- ─────────────────────────────────────────────────────────────
    -- convergence_snapshots table
    -- ─────────────────────────────────────────────────────────────
    CREATE TABLE IF NOT EXISTS convergence_snapshots (
      id                    UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      goal_space_id         UUID NOT NULL REFERENCES nodes(id) ON DELETE CASCADE,
      score                 FLOAT NOT NULL,
      factor_breakdown      JSONB NOT NULL DEFAULT '{}',
      node_count_at_snapshot INT NOT NULL DEFAULT 0,
      computed_at           TIMESTAMPTZ NOT NULL DEFAULT NOW()
    );

    -- Index: latest snapshot for a goal space (trajectory badge, Phase 5)
    CREATE INDEX IF NOT EXISTS idx_convergence_snapshots_goal_space_computed
      ON convergence_snapshots(goal_space_id, computed_at DESC);

    -- Index: all snapshots by computed_at (sparkline queries, Phase 5)
    CREATE INDEX IF NOT EXISTS idx_convergence_snapshots_computed
      ON convergence_snapshots(computed_at DESC);

    -- ─────────────────────────────────────────────────────────────
    -- Row Level Security (permissive model — matches all other tables)
    -- ─────────────────────────────────────────────────────────────
    ALTER TABLE convergence_snapshots ENABLE ROW LEVEL SECURITY;

    CREATE POLICY "Authenticated users can read convergence_snapshots"
      ON convergence_snapshots FOR SELECT TO authenticated USING (true);

    CREATE POLICY "Authenticated users can manage convergence_snapshots"
      ON convergence_snapshots FOR ALL TO authenticated USING (true) WITH CHECK (true);
    ```
